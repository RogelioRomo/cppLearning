| >>=      | lhs >>= rhs; | lhs = lhs >> (rhs); |

| <<=      | lhs <<= rhs; | lhs = lhs << (rhs); |

| &=       | lhs &= rhs; | lhs = lhs & (rhs);   |
-------------------------------------------------
| ^=       | lhs ^= rhs; | lhs = lhs ^ (rhs);   |

| |=       | lhs |= rhs; | lhs = lhs | (rhs);   |
